package bullsAndCows.test;

import java.io.*;

public class Test {
    private static void readFile() throws IOException {
        BufferedReader bufferedReader = new BufferedReader(new FileReader("saeed.txt"));
        PrintWriter printWriter = new PrintWriter(new FileWriter("copySaeed.txt"));
        String str;
        while ((str = bufferedReader.readLine()) != null) {
            printWriter.println(str);
        }
        bufferedReader.close();
        printWriter.close();
        System.out.println("File copied!");
    }
}
